package com.example.buttonevents;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {
    private Button button1, button2, button3;
    private TextView resultText;
    private int clickCount = 0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        button1 = findViewById(R.id.button1);
        button2 = findViewById(R.id.button2);
        button3 = findViewById(R.id.button3);
        resultText = findViewById(R.id.resultText);
        
        // Method 1: Anonymous inner class
        button1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                clickCount++;
                resultText.setText("Button 1 clicked " + clickCount + " times");
                Toast.makeText(MainActivity.this, "Button 1 clicked!", Toast.LENGTH_SHORT).show();
            }
        });
        
        // Method 2: Lambda expression
        button2.setOnClickListener(v -> {
            resultText.setText("Button 2 clicked using Lambda!");
            Toast.makeText(this, "Lambda click!", Toast.LENGTH_SHORT).show();
        });
        
        // Method 3: Method reference
        button3.setOnClickListener(this::onButton3Click);
    }
    
    public void onButton3Click(View view) {
        resultText.setText("Button 3 clicked using method reference!");
        Toast.makeText(this, "Method reference click!", Toast.LENGTH_SHORT).show();
    }
}
6. Registration Form
Complete form with validation
// MainActivity.java
package com.example.registrationform;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.*;

public class MainActivity extends AppCompatActivity {
    private EditText nameEdit, emailEdit, phoneEdit, passwordEdit;
    private RadioGroup genderGroup;
    private CheckBox termsCheck;
    private Spinner citySpinner;
    private Button submitBtn, clearBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        initViews();
        setupSpinner();
        
        submitBtn.setOnClickListener(v -> {
            if (validateForm()) {
                submitForm();
            }
        });
        
        clearBtn.setOnClickListener(v -> clearForm());
    }
    
    private void initViews() {
        nameEdit = findViewById(R.id.nameEdit);
        emailEdit = findViewById(R.id.emailEdit);
        phoneEdit = findViewById(R.id.phoneEdit);
        passwordEdit = findViewById(R.id.passwordEdit);
        genderGroup = findViewById(R.id.genderGroup);
        termsCheck = findViewById(R.id.termsCheck);
        citySpinner = findViewById(R.id.citySpinner);
        submitBtn = findViewById(R.id.submitBtn);
        clearBtn = findViewById(R.id.clearBtn);
    }
    
    private void setupSpinner() {
        String[] cities = {"Select City", "Mumbai", "Delhi", "Bangalore", "Chennai", "Kolkata"};
        ArrayAdapter adapter = new ArrayAdapter<>(this, 
            android.R.layout.simple_spinner_item, cities);
        adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
        citySpinner.setAdapter(adapter);
    }
    
    private boolean validateForm() {
        if (nameEdit.getText().toString().trim().isEmpty()) {
            nameEdit.setError("Name is required");
            nameEdit.requestFocus();
            return false;
        }
        
        String email = emailEdit.getText().toString().trim();
        if (email.isEmpty() || !android.util.Patterns.EMAIL_ADDRESS.matcher(email).matches()) {
            emailEdit.setError("Valid email is required");
            emailEdit.requestFocus();
            return false;
        }
        
        String phone = phoneEdit.getText().toString().trim();
        if (phone.isEmpty() || phone.length() != 10) {
            phoneEdit.setError("Valid 10-digit phone number required");
            phoneEdit.requestFocus();
            return false;
        }
        
        if (passwordEdit.getText().toString().length() < 6) {
            passwordEdit.setError("Password must be at least 6 characters");
            passwordEdit.requestFocus();
            return false;
        }
        
        if (genderGroup.getCheckedRadioButtonId() == -1) {
            Toast.makeText(this, "Please select gender", Toast.LENGTH_SHORT).show();
            return false;
        }
        
        if (citySpinner.getSelectedItemPosition() == 0) {
            Toast.makeText(this, "Please select city", Toast.LENGTH_SHORT).show();
            return false;
        }
        
        if (!termsCheck.isChecked()) {
            Toast.makeText(this, "Please accept terms and conditions", Toast.LENGTH_SHORT).show();
            return false;
        }
        
        return true;
    }
    
    private void submitForm() {
        String name = nameEdit.getText().toString();
        String email = emailEdit.getText().toString();
        String phone = phoneEdit.getText().toString();
        
        RadioButton selectedGender = findViewById(genderGroup.getCheckedRadioButtonId());
        String gender = selectedGender.getText().toString();
        
        String city = citySpinner.getSelectedItem().toString();
        
        String message = "Registration Successful!\n\n" +
                        "Name: " + name + "\n" +
                        "Email: " + email + "\n" +
                        "Phone: " + phone + "\n" +
                        "Gender: " + gender + "\n" +
                        "City: " + city;
        
        Toast.makeText(this, message, Toast.LENGTH_LONG).show();
    }
    
    private void clearForm() {
        nameEdit.setText("");
        emailEdit.setText("");
        phoneEdit.setText("");
        passwordEdit.setText("");
        genderGroup.clearCheck();
        citySpinner.setSelection(0);
        termsCheck.setChecked(false);
    }
}

// MainActivity.java
package com.example.calculator;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.*;

public class MainActivity extends AppCompatActivity {
    private EditText num1Edit, num2Edit;
    private TextView resultText;
    private Button addBtn, subBtn, mulBtn, divBtn, clearBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        initViews();
        setClickListeners();
    }
    
    private void initViews() {
        num1Edit = findViewById(R.id.num1Edit);
        num2Edit = findViewById(R.id.num2Edit);
        resultText = findViewById(R.id.resultText);
        addBtn = findViewById(R.id.addBtn);
        subBtn = findViewById(R.id.subBtn);
        mulBtn = findViewById(R.id.mulBtn);
        divBtn = findViewById(R.id.divBtn);
        clearBtn = findViewById(R.id.clearBtn);
    }
    
    private void setClickListeners() {
        addBtn.setOnClickListener(v -> performOperation("+"));
        subBtn.setOnClickListener(v -> performOperation("-"));
        mulBtn.setOnClickListener(v -> performOperation("*"));
        divBtn.setOnClickListener(v -> performOperation("/"));
        clearBtn.setOnClickListener(v -> clearAll());
    }
    
    private void performOperation(String operation) {
        try {
            double num1 = Double.parseDouble(num1Edit.getText().toString());
            double num2 = Double.parseDouble(num2Edit.getText().toString());
            double result = 0;
            
            switch (operation) {
                case "+":
                    result = num1 + num2;
                    break;
                case "-":
                    result = num1 - num2;
                    break;
                case "*":
                    result = num1 * num2;
                    break;
                case "/":
                    if (num2 != 0) {
                        result = num1 / num2;
                    } else {
                        Toast.makeText(this, "Cannot divide by zero!", Toast.LENGTH_SHORT).show();
                        return;
                    }
                    break;
            }
            
            resultText.setText("Result: " + result);
            
        } catch (NumberFormatException e) {
            Toast.makeText(this, "Please enter valid numbers", Toast.LENGTH_SHORT).show();
        }
    }
    
    private void clearAll() {
        num1Edit.setText("");
        num2Edit.setText("");
        resultText.setText("Result: ");
    }
}
8. ListView Example
ListView with custom adapter
// MainActivity.java
package com.example.listview;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.*;
import java.util.ArrayList;
import java.util.Arrays;

public class MainActivity extends AppCompatActivity {
    private ListView listView;
    private ArrayList items;
    private ArrayAdapter adapter;
    private EditText addItemEdit;
    private Button addBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        initViews();
        setupListView();
        
        addBtn.setOnClickListener(v -> addItem());
    }
    
    private void initViews() {
        listView = findViewById(R.id.listView);
        addItemEdit = findViewById(R.id.addItemEdit);
        addBtn = findViewById(R.id.addBtn);
    }
    
    private void setupListView() {
        String[] itemArray = {"Apple", "Banana", "Orange", "Mango", "Grapes", 
                             "Pineapple", "Strawberry", "Watermelon"};
        items = new ArrayList<>(Arrays.asList(itemArray));
        
        adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, items);
        listView.setAdapter(adapter);
        
        // Item click listener
        listView.setOnItemClickListener((parent, view, position, id) -> {
            String selectedItem = items.get(position);
            Toast.makeText(this, "Selected: " + selectedItem, Toast.LENGTH_SHORT).show();
        });
        
        // Long click to delete
        listView.setOnItemLongClickListener((parent, view, position, id) -> {
            String removedItem = items.remove(position);
            adapter.notifyDataSetChanged();
            Toast.makeText(this, "Removed: " + removedItem, Toast.LENGTH_SHORT).show();
            return true;
        });
    }
    
    private void addItem() {
        String newItem = addItemEdit.getText().toString().trim();
        if (!newItem.isEmpty()) {
            items.add(newItem);
            adapter.notifyDataSetChanged();
            addItemEdit.setText("");
            Toast.makeText(this, "Added: " + newItem, Toast.LENGTH_SHORT).show();
        } else {
            Toast.makeText(this, "Please enter item name", Toast.LENGTH_SHORT).show();
        }
    }
}

// MainActivity.java
package com.example.intentexample;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.os.Bundle;
import android.widget.*;

public class MainActivity extends AppCompatActivity {
    private EditText nameEdit, messageEdit;
    private Button sendBtn, implicitBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        nameEdit = findViewById(R.id.nameEdit);
        messageEdit = findViewById(R.id.messageEdit);
        sendBtn = findViewById(R.id.sendBtn);
        implicitBtn = findViewById(R.id.implicitBtn);
        
        // Explicit Intent
        sendBtn.setOnClickListener(v -> {
            String name = nameEdit.getText().toString();
            String message = messageEdit.getText().toString();
            
            Intent intent = new Intent(MainActivity.this, SecondActivity.class);
            intent.putExtra("NAME", name);
            intent.putExtra("MESSAGE", message);
            startActivity(intent);
        });
        
        // Implicit Intent
        implicitBtn.setOnClickListener(v -> {
            Intent intent = new Intent(Intent.ACTION_SEND);
            intent.setType("text/plain");
            intent.putExtra(Intent.EXTRA_TEXT, "Hello from my app!");
            startActivity(Intent.createChooser(intent, "Share via"));
        });
    }
}

                return true;

// MainActivity.java
package com.example.sharedpreferences;

import androidx.appcompat.app.AppCompatActivity;
import android.content.SharedPreferences;
import android.os.Bundle;
import android.widget.*;

public class MainActivity extends AppCompatActivity {
    private EditText nameEdit, emailEdit;
    private CheckBox rememberCheck;
    private Button saveBtn, loadBtn, clearBtn;
    private SharedPreferences sharedPreferences;
    private static final String PREF_NAME = "UserPrefs";

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        initViews();
        sharedPreferences = getSharedPreferences(PREF_NAME, MODE_PRIVATE);
        
        saveBtn.setOnClickListener(v -> savePreferences());
        loadBtn.setOnClickListener(v -> loadPreferences());
        clearBtn.setOnClickListener(v -> clearPreferences());
        
        // Auto-load on startup
        loadPreferences();
    }
    
    private void initViews() {
        nameEdit = findViewById(R.id.nameEdit);
        emailEdit = findViewById(R.id.emailEdit);
        rememberCheck = findViewById(R.id.rememberCheck);
        saveBtn = findViewById(R.id.saveBtn);
        loadBtn = findViewById(R.id.loadBtn);
        clearBtn = findViewById(R.id.clearBtn);
    }
    
    private void savePreferences() {
        String name = nameEdit.getText().toString();
        String email = emailEdit.getText().toString();
        boolean remember = rememberCheck.isChecked();
        
        SharedPreferences.Editor editor = sharedPreferences.edit();
        editor.putString("name", name);
        editor.putString("email", email);
        editor.putBoolean("remember", remember);
        editor.apply();
        
        Toast.makeText(this, "Preferences saved!", Toast.LENGTH_SHORT).show();
    }
    
    private void loadPreferences() {
        String name = sharedPreferences.getString("name", "");
        String email = sharedPreferences.getString("email", "");
        boolean remember = sharedPreferences.getBoolean("remember", false);
        
        nameEdit.setText(name);
        emailEdit.setText(email);
        rememberCheck.setChecked(remember);
        
        Toast.makeText(this, "Preferences loaded!", Toast.LENGTH_SHORT).show();
    }
    
    private void clearPreferences() {
        SharedPreferences.Editor editor = sharedPreferences.edit();
        editor.clear();
        editor.apply();
        
        nameEdit.setText("");
        emailEdit.setText("");
        rememberCheck.setChecked(false);
        
        Toast.makeText(this, "Preferences cleared!", Toast.LENGTH_SHORT).show();
    }
}
